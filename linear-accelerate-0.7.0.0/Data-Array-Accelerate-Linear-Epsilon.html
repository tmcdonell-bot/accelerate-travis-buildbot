class Num a => Epsilon a where Source #

Provides a fairly subjective test to see if a quantity is near zero.

>>> nearZero (1e-11 :: Exp Double)
False

>>> nearZero (1e-17 :: Exp Double)
True

>>> nearZero (1e-5 :: Exp Float)
False

>>> nearZero (1e-7 :: Exp Float)
True
