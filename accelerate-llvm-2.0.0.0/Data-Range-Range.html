bisect :: Range -> (Range, Range) Source

O(1). Split an interval into two roughly equally sized ranges. If the interval is odd then the first interval gets the extra element.

take :: Int -> Range -> Range Source

O(1). Return the first n elements of the range, or the range itself if n > size.

splitAt :: Int -> Range -> (Range, Range) Source

O(1). A tuple where the first element is the first n elements of the range, and the second is the remainder of the list (if any).

merge :: Range -> Range -> Maybe Range Source

If the two ranges are adjacent, return one combined range. The ranges must not be empty.

append :: Seq Range -> Range -> Seq Range Source

O(1). Add a new range to the end of the given sequence. We assume that ranges are non-overlapping and non-empty. If the new range is adjacent to the last range on the sequence, the ranges are appended.

compress :: Seq Range -> Seq Range Source

O(n log n). Compress the given ranges into the fewest number of sections as possible. The ranges must not be empty.
