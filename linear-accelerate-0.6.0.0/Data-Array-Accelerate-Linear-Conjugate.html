    class Num a => Conjugate a where #

    An involutive ring

    Methods

    conjugate :: a -> a #

    Conjugate a value. This defaults to the trivial involution.

    >>> conjugate (1 :+ 2)
1.0 :+ (-2.0)

    >>> conjugate 1
1
