    class Num a => Epsilon a where Source #

    Provides a fairly subjective test to see if a quantity is near zero.

    >>> nearZero (1e-11 :: Double)
False

    >>> nearZero (1e-17 :: Double)
True

    >>> nearZero (1e-5 :: Float)
False

    >>> nearZero (1e-7 :: Float)
True
